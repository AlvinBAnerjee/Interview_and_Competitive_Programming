#include<iostream>

long long linesPassingThroughOrigin(std::vector<std::vector<long long>> &points , int n)
{
    std::unordered_map<float,long long> umap;
    for(int i = 0 ; i < n ; i++)
    {
        if(umap.find((points[i][1])/(double)points[i][0]) == umap.end())
        {
            umap[(points[i][1])/(double)points[i][0]] = 1;
        }
        else
        {
            umap[(points[i][1])/(double)points[i][0]]++;
        }
    }
    long long count = 0;
    for(auto i : umap)
    {
        count += (i.second*(i.second-1))/2;
    }
    return count;
}

#include <bits/stdc++.h>
using namespace std;

int solve(int n, int x, vector<int>& arr){
    if(x > n) return 0;
    priority_queue<int, vector<int>, greater<int>>pq;
    for(auto it: arr) pq.push(it);
    while(pq.size() != x){
        int x = pq.top();
        pq.pop();
        int y = pq.top();
        pq.pop();
        pq.push(x + y);
    }
    int low = 0, high = *max_element(arr.begin(), arr.end()), mid, res = -1;
    while(low <= high){
        mid = low + (high - low) / 2;
        if(pq.top() >= mid){
            res = mid;
            low = mid + 1;
        }
        else high = mid - 1;
    }
    return res;
}

    return 0;
}

bool canDistributeXdistinctCandies(std::vector<int> candies , int x , int n , long long k)
{
    int count = 0;
    long long sum = 0;
    for(int i = 0 ; i < n ; i++)
    {

        if(candies[i] >= k)
        {
            count++;
        }
        else if(sum+candies[i] >= k)
        {
            sum = sum+candies[i]-k;
            count++;
        }
        else
        {
            sum += candies[i];
        }
        if(count >= x)
            return true;
    }
    return false;
}
long long maxFreindsCanBeInvited(std::vector<int> &candies , int n , int x)
{
    std::sort(candies.rbegin(),candies.rend());
    long long start = 1;
    long long end = 100000000000;
    long long res = 0;
    long long mid = 0;
    while(start <= end)
    {
        mid = (start+end)/2;
        if(canDistributeXdistinctCandies(candies,x,n,mid) == true)
        {
            if(mid > res)
                res = mid;
            start = mid+1;
        }
        else
        {
            end = mid-1;
        }
    }
    return res;
}
